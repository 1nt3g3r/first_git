status - стан репозиторію
init - створення репозиторію
add <назва файлу> - додати файл до репозиторію
git config user.email
git config user.name
git log - історія комітів
git branch - список гілок + активна гілка
git checkout -b <гілка> - створення нової гілки і перемикання
git diff <коміт 1> <коміт 2> [файл] - зміни між комітами
git remote -v - список віддалених репозиторіїв

git push <віддалений репозиторій> <гілка> - заливає ваш код у віддалений репозиторій
git pull <віддалений репозиторій> <гілка> - витягує вказану гілку з віддаленого репозиторію у ваш репозиторій
